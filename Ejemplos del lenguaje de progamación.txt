	definir Espacios entero;

	definir K entero;
	K <- 0;

		imprimir " ";
	
	hasta(K >= Espacios);

	definir Ancho = (2*I)-1;

	mientras(J < Ancho)

		imprimir "*";
		J <- I+1;

	repetir;

	I <- I+1;
