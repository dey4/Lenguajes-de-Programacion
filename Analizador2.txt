void onError()
{
	if (!error) {
		fprintf(salida, "linea %d, ERROR: %s", counter, yytext);
	}
	else {
		fprintf(salida, "%s", yytext);
	}

	error = 1;
}

void endError()
{
	if (error) {
		fprintf(salida, "\n");
		error = 0;
	}
}

int yywrap()
{
        return 1;
}

}
